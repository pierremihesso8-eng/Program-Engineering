        else:
            print("Томаты еще не созрели! Нужно продолжать ухаживать.")
    
    @staticmethod
    def knowledge_base():
        print("Справка по садоводству: Помидоры созревают через несколько стадий: отсутствует, цветение, зеленый, красный.")
